<script>

    // ブラウザが読み込まれてから実行

    window.onload = function() {

        var vals = [1, 1, 1, 1, 1];

        var keeps = [false, false, false, false, false];

        var left = 3;

        var marks = ["", "⚀", "⚁", "⚂", "⚃", "⚄", "⚅"];


        var rollBtn = document.getElementById("roll-btn");

        var countText = document.getElementById("count");


        // 振るボタンの処理

        rollBtn.onclick = function() {

            if (left > 0) {

                for (var i = 0; i < 5; i++) {

                    if (keeps[i] === false) {

                        vals[i] = Math.floor(Math.random() * 6) + 1;

                        document.getElementById("d" + i).innerHTML = marks[vals[i]];

                    }

                }

                left--;

                countText.innerHTML = left;

                if (left === 0) rollBtn.disabled = true;

            }

        };


        // サイコロクリック（キープ）の処理

        function setClick(n) {

            var el = document.getElementById("d" + n);

            el.onclick = function() {

                if (left < 3) {

                    keeps[n] = !keeps[n];

                    this.className = keeps[n] ? "die kept" : "die";

                }

            };

        }


        for (var i = 0; i < 5; i++) {

            setClick(i);

        }

    };

</script>
